function elements(param) {
 return new function() {
  this.me = param;

  this.elements = function(arg) {
   return elements(arg)
  }

  this.first = function() {
	var k = this.me;
	
	return new function() {
			this.e = k[0];

   			this.text = function() {
   				return this.e.textContent;
   			}
   }
  }
 }
}
